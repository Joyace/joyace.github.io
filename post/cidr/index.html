最近由于需要用iptables设置一些过滤规则，也有bind里配置acl的时候有合并地址段的需求。看了下cpan里有个现成的模块，几行代码就可以做这个合并工作。

#!/usr/bin/perl
use strict;
use warnings;
use Net::CIDR::Lite;
my $cidr = Net::CIDR::Lite->new;
while (my $line=<>) {
    $cidr->add($line);
}
foreach my $line( @{$cidr->list} ) {
    print "$line\n";
}

#!/usr/bin/env python
import cidrize
import sys
import string
def ipmerge(str):
    obj=cidrize.optimize_network_range(str)
    return ";\n".join(cidrize.output_str(obj).split(", "))+";"
def main():
    if len(sys.argv)!=2:
        print "./ipmerge.py ipfile.txt"
        sys.exit(1)
    try:
        fd=open(sys.argv[1],"r")
    except IOError as e:
        print "open ip file failed,",e
    lines=""
    for line in fd:
        lines += line.rstrip(";\n")+","
    lines=lines.rstrip(",")
    print ipmerge(lines)
if __name__ == "__main__":
    main()
